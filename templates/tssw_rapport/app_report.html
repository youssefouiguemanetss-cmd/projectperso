Authentication System

Flask-Login Authentication Flow

How Login Works

  1. 1

    User submits username and password via login form

  2. 2

    load_users_from_file() reads users.txt and parses credentials

  3. 3

    Password is compared (plaintext in file)

  4. 4

    login_user(user, remember=True) creates session

  5. 5

    User object with entity and permissions stored in session

app.py - Login Route Python 
@app.route('/login', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        username = request.form.get('username')
        password = request.form.get('password')
        
        users = load_users_from_file()
        for user_data in users:
            if user_data['username'] == username:
                if user_data['password'] == password:
                    user = User(user_data)
                    login_user(user, remember=True)
                    return redirect(url_for('services'))
        
        flash('Invalid credentials', 'error')
    return render_template('login.html')

IMAP Connection Python 
import imaplib

def connect_to_gmail(email, app_password):
    # Create IMAP4 client with SSL
    imap = imaplib.IMAP4_SSL(
        'imap.gmail.com', 
        993
    )
    
    # Authenticate with App Password
    imap.login(email, app_password)
    
    # Select inbox folder
    imap.select('INBOX')
    
    # Get last 20 UIDs
    status, messages = imap.uid('search', None, 'ALL')
    uids = messages[0].split()[-20:]
    
    return imap

connection_manager.py Python 
class GmailConnectionManager:
    def __init__(self):
        self.entity_connections = {}
        self.entity_user_count = {}
        self.lock = threading.Lock()
    
    def user_login(self, entity):
        with self.lock:
            if entity not in self.entity_connections:
                self._create_connections(entity)
            self.entity_user_count[entity] += 1
